<script>

(function(){

  const MIN_INTERVAL = 15, MAX_INTERVAL = 120;

  const REMIND_REPEAT_MS = 5*60*1000; // re-notify every 5 min while overdue

 

  let state = {

    water: { interval: 45, lastDone: Date.now(), count: 0, lastNotified: 0 },

    move:  { interval: 60, lastDone: Date.now(), count: 0, lastNotified: 0 },

    date: todayStr(),

    running: true

  };

 

  function todayStr(){

    const d = new Date();

    return d.getFullYear()+'-'+(d.getMonth()+1)+'-'+d.getDate();

  }

 

  function loadState(){

    try{

      const raw = localStorage.getItem('hydration-tool-state');

      if(raw){

        const parsed = JSON.parse(raw);

        state = Object.assign(state, parsed);

        if(state.date !== todayStr()){

          state.water.count = 0;

          state.move.count = 0;

          state.date = todayStr();

        }

      }

    }catch(e){ /* no saved state yet, or storage unavailable */ }

    render();

  }

 

  function saveState(){

    try{ localStorage.setItem('hydration-tool-state', JSON.stringify(state)); }

    catch(e){ console.error('save failed', e); }

  }

 

  function fmt(ms){

    const overdue = ms < 0;

    const abs = Math.abs(ms);

    const total = Math.floor(abs/1000);

    const m = Math.floor(total/60);

    const s = total%60;

    return (overdue?'-':'') + String(m).padStart(2,'0') + ':' + String(s).padStart(2,'0');

  }

 

  function remaining(key){

    const s = state[key];

    return s.interval*60000 - (Date.now() - s.lastDone);

  }

 

  function notify(title, body){

    if('Notification' in window && Notification.permission === 'granted'){

      try{ new Notification(title, {body, silent:false}); }catch(e){}

    }

    beep();

  }

 

  function beep(){

    try{

      const ctx = new (window.AudioContext || window.webkitAudioContext)();

      const o = ctx.createOscillator();

      const g = ctx.createGain();

      o.type = 'sine';

      o.frequency.value = 660;

      g.gain.value = 0.0001;

      o.connect(g); g.connect(ctx.destination);

      const now = ctx.currentTime;

      g.gain.exponentialRampToValueAtTime(0.15, now+0.02);

      g.gain.exponentialRampToValueAtTime(0.0001, now+0.5);

      o.start(now); o.stop(now+0.55);

    }catch(e){}

  }

 

  function doDone(key){

    state[key].lastDone = Date.now();

    state[key].count += 1;

    state[key].lastNotified = 0;

    saveState();

    render();

  }

 

  document.getElementById('waterDone').onclick = ()=>doDone('water');

  document.getElementById('moveDone').onclick = ()=>doDone('move');

 

  document.querySelectorAll('[data-adj]').forEach(btn=>{

    btn.onclick = ()=>{

      const key = btn.dataset.adj;

      const delta = parseInt(btn.dataset.delta,10);

      let v = state[key].interval + delta;

      v = Math.max(MIN_INTERVAL, Math.min(MAX_INTERVAL, v));

      state[key].interval = v;

      saveState();

      render();

    };

  });

 

  document.getElementById('toggleRun').onclick = ()=>{

    state.running = !state.running;

    if(state.running){

      state.water.lastDone = Date.now() - Math.max(0,(state.water.interval*60000 - remaining('water')));

    }

    saveState();

    render();

  };

 

  document.getElementById('askNotif').onclick = async ()=>{

    if(!('Notification' in window)){

      alert('当前浏览器不支持系统通知，页面内提醒仍会正常工作。');

      return;

    }

    const perm = await Notification.requestPermission();

    render();

  };

 

  function updatePlant(){

    const waterHealth = Math.max(0, Math.min(1, remaining('water') / (state.water.interval*60000)));

    const moveHealth = Math.max(0, Math.min(1, remaining('move') / (state.move.interval*60000)));

 

    const healthy = [70,122,87]; // green

    const wilted = [156,139,78]; // olive

    const mix = (a,b,t)=> Math.round(a + (b-a)*(1-t));

    const leafColor = `rgb(${mix(healthy[0],wilted[0],waterHealth)},${mix(healthy[1],wilted[1],waterHealth)},${mix(healthy[2],wilted[2],waterHealth)})`;

 

    ['1','2','3','4'].forEach(n=>{

      document.getElementById('leafPath'+n).setAttribute('fill', leafColor);

    });

 

    const droop = (1-moveHealth) * 30; // degrees

    document.getElementById('leaf1').style.transform = `rotate(${droop}deg)`;

    document.getElementById('leaf2').style.transform = `rotate(${droop*0.8}deg)`;

    document.getElementById('leaf3').style.transform = `rotate(${-droop}deg)`;

    document.getElementById('leaf4').style.transform = `rotate(${-droop*0.8}deg)`;

    document.getElementById('stem').setAttribute('stroke', mix(76,120,waterHealth) > 100 ? '#4C7C57' : '#8B8555');

 

    const overall = Math.min(waterHealth, moveHealth);

    const bud = document.getElementById('bud');

    bud.setAttribute('fill', '#E8A33D');

    bud.setAttribute('opacity', overall > 0.75 ? 1 : 0);

 

    const statusText = document.getElementById('statusText');

    const statusSub = document.getElementById('statusSub');

    if(overall > 0.75){ statusText.textContent = '元气满满'; statusSub.textContent = '继续保持'; }

    else if(waterHealth <= 0 && moveHealth <= 0){ statusText.textContent = '又渴又想动'; statusSub.textContent = '两件事都该做啦'; }

    else if(waterHealth <= 0){ statusText.textContent = '有点渴了'; statusSub.textContent = '去喝口水吧'; }

    else if(moveHealth <= 0){ statusText.textContent = '坐太久了'; statusSub.textContent = '起来走两步'; }

    else if(overall < 0.35){ statusText.textContent = '快撑不住了'; statusSub.textContent = '提醒快到了'; }

    else { statusText.textContent = '状态还不错'; statusSub.textContent = ''; }

  }

 

  function render(){

    const wr = remaining('water');

    document.getElementById('waterTimer').textContent = fmt(wr);

    document.getElementById('waterTimer').classList.toggle('overdue', wr<0);

    document.getElementById('waterLabel').textContent = wr<0 ? '已超时' : '距下一次提醒';

    document.getElementById('waterInterval').textContent = state.water.interval+' 分钟';

    document.getElementById('waterToday').textContent = '今日 '+state.water.count+' 次';

    document.getElementById('cardWater').classList.toggle('due', wr<0);

 

    const mr = remaining('move');

    document.getElementById('moveTimer').textContent = fmt(mr);

    document.getElementById('moveTimer').classList.toggle('overdue', mr<0);

    document.getElementById('moveLabel').textContent = mr<0 ? '已超时' : '距下一次提醒';

    document.getElementById('moveInterval').textContent = state.move.interval+' 分钟';

    document.getElementById('moveToday').textContent = '今日 '+state.move.count+' 次';

    document.getElementById('cardMove').classList.toggle('due', mr<0);

 

    document.getElementById('toggleRun').textContent = state.running ? '暂停提醒' : '继续提醒';

    document.getElementById('toggleRun').classList.toggle('off', !state.running);

 

    const notifBtn = document.getElementById('askNotif');

    if('Notification' in window && Notification.permission === 'granted'){

      notifBtn.textContent = '系统通知已开启';

      notifBtn.classList.add('off');

    } else {

      notifBtn.textContent = '开启系统通知';

      notifBtn.classList.remove('off');

    }

 

    updatePlant();

  }

 

  function tick(){

    if(!state.running) return;

    ['water','move'].forEach(key=>{

      const r = remaining(key);

      if(r < 0){

        const sinceNotify = Date.now() - (state[key].lastNotified || 0);

        if(sinceNotify > REMIND_REPEAT_MS || state[key].lastNotified === 0){

          state[key].lastNotified = Date.now();

          if(key === 'water') notify('该喝水啦 💧', '你的植物有点渴了，去倒杯水吧');

          else notify('该起身活动啦 🚶', '坐了挺久了，起来走两步、伸个懒腰');

        }

      }

    });

    if(state.date !== todayStr()){

      state.water.count = 0;

      state.move.count = 0;

      state.date = todayStr();

      saveState();

    }

    render();

  }

 

  window.addEventListener('beforeunload', saveState);

  loadState();

  setInterval(tick, 1000);

})();

</script>
